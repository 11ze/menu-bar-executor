### Task 1: 修改 NotificationManager 添加执行结果参数

**Files:**
- Modify: `Sources/App/NotificationManager.swift:19-47`

**Step 1: 修改 showSuccess 方法**

```swift
func showSuccess(commandName: String, output: String?) {
    let content = UNMutableNotificationContent()
    content.title = "命令执行成功"

    // 显示命令名称和执行结果（如果有输出）
    if let output = output, !output.isEmpty {
        // 限制输出长度，避免通知内容过长
        let truncatedOutput = output.count > 100 ? String(output.prefix(100)) + "..." : output
        content.body = "\(commandName)\n\(truncatedOutput)"
    } else {
        content.body = commandName
    }

    content.sound = .default

    let request = UNNotificationRequest(
        identifier: UUID().uuidString,
        content: content,
        trigger: nil
    )

    UNUserNotificationCenter.current().add(request)
}
```

**Step 2: 修改 showFailure 方法**

```swift
func showFailure(commandName: String, error: String, output: String?) {
    let content = UNMutableNotificationContent()
    content.title = "命令执行失败"

    // 显示命令名称、错误信息和执行结果（如果有）
    var body = "\(commandName): \(error)"
    if let output = output, !output.isEmpty {
        let truncatedOutput = output.count > 100 ? String(output.prefix(100)) + "..." : output
        body += "\n\(truncatedOutput)"
    }

    content.body = body
    content.sound = .default

    let request = UNNotificationRequest(
        identifier: UUID().uuidString,
        content: content,
        trigger: nil
    )

    UNUserNotificationCenter.current().add(request)
}
```

### Task 2: 修改 CommandsManager 传递执行结果

**Files:**
- Modify: `Sources/App/CommandsManager.swift:31-43`

**Step 1: 修改 execute 方法调用**

```swift
func execute(_ command: Command) {
    executor.execute(command: command) { [weak self] success, output in
        guard let self = self else { return }

        if command.notification {
            if success {
                self.notificationManager.showSuccess(commandName: command.name, output: output)
            } else {
                let errorMsg = output ?? "未知错误"
                self.notificationManager.showFailure(commandName: command.name, error: errorMsg, output: output)
            }
        }
    }
}
```
